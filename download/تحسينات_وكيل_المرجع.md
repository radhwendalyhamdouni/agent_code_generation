### المرحلة الأولى: محرك AI حقيقي

#### 1.1 دمج LLM حقيقي مع z-ai-web-dev-sdk

```typescript
// src/lib/ai-engine.ts
import ZAI from 'z-ai-web-dev-sdk';

interface AIEngine {
  generateCode(prompt: string): Promise<string>;
  explainCode(code: string): Promise<string>;
  fixCode(code: string, error: string): Promise<string>;
  optimizeCode(code: string): Promise<string>;
}

export class AlmarjaaAIEngine implements AIEngine {
  private zai: any;
  
  async initialize() {
    this.zai = await ZAI.create();
  }
  
  async generateCode(prompt: string): Promise<string> {
    const systemPrompt = `
أنت مبرمج خبير بلغة المرجع العربية.
قواعد اللغة:
- متغير اسم = قيمة؛
- دالة اسم(معاملات) { ... }
- إذا شرط { } وإلا إذا { } وإلا { }
- لكل عنصر في [قائمة] { }
- اطبع("نص")؛
- أرجع قيمة؛

الدوال المتاحة: طول، جذر، أس، مطلق، وقت
أرجع كوداً صالحاً فقط.
    `;
    
    const completion = await this.zai.chat.completions.create({
      messages: [
        { role: 'system', content: systemPrompt },
        { role: 'user', content: prompt }
      ]
    });
    
    return completion.choices[0]?.message?.content || '';
  }
  
  async fixCode(code: string, error: string): Promise<string> {
    const prompt = `
الكود التالي به خطأ:
${code}

الخطأ:
${error}

أصلح الكود وأرجع النسخة المصححة فقط.
    `;
    
    return this.generateCode(prompt);
  }
}
```

#### 2.2 تنفيذ Web-based GUI

```typescript
// src/app/api/gui/route.ts
import { NextRequest, NextResponse } from 'next/server';

export async function POST(request: NextRequest) {
  const { code } = await request.json();
  
  // تحويل كود المرجع GUI إلى HTML/CSS/JS
  const html = convertAlmarjaaGUIToHTML(code);
  
  return NextResponse.json({ html });
}

function convertAlmarjaaGUIToHTML(code: string): string {
  // تحويل: واجهة.زر.جديد({نص: "مرحباً"})
  // إلى: <button>مرحباً</button>
  
  return code
    .replace(/واجهة\.زر\.جديد\({\s*نص:\s*"([^"]+)"\s*}\)/g, '<button>$1</button>')
    .replace(/واجهة\.نص\.جديد\({\s*نص:\s*"([^"]+)"\s*}\)/g, '<p>$1</p>')
    .replace(/واجهة\.مدخل\.جديد\({\s*نص:\s*"([^"]+)"\s*}\)/g, '<input placeholder="$1"/>');
}
```

// src/components/agent/code-editor.tsx
import Editor from '@monaco-editor/react';

export function AlmarjaaEditor({ code, onChange }: Props) {
  return (
    <Editor
      height="100%"
      defaultLanguage="almarjaa"
      value={code}
      onChange={onChange}
      theme="almarjaa-dark"
      options={{
        direction: 'rtl',
        fontFamily: 'Tajawal, monospace',
        fontSize: 14,
        minimap: { enabled: true },
        formatOnPaste: true,
        autoIndent: 'advanced',
      }}
      beforeMount={(monaco) => {
        // تسجيل لغة المرجع
        monaco.languages.register({ id: 'almarjaa' });
        
        monaco.languages.setMonarchTokensProvider('almarjaa', {
          keywords: ['متغير', 'دالة', 'إذا', 'وإلا', 'لكل', 'أرجع', 'اطبع'],
          operators: ['+', '-', '*', '/', '==', '!=', '>', '<', '>=', '<='],
          symbols: /[=><!~?:&|+\-*\/\^%]+/,
          tokenizer: {
            root: [
              [/[أ-ي]+/, 'identifier.arabic'],
              [/\d+/, 'number'],
              [/"([^"]*)"/, 'string'],
              [/\/\//, 'comment'],
            ]
          }
        });
      }}
    />
  );
}

#### 3.2 نظام التصحيح (Debugger)

```typescript
// src/lib/debugger.ts
export class AlmarjaaDebugger {
  private breakpoints: number[] = [];
  private variables: Map<string, any> = new Map();
  
  addBreakpoint(line: number) {
    this.breakpoints.push(line);
  }
  
  async stepThrough(code: string): Promise<DebugStep[]> {
    const steps: DebugStep[] = [];
    const lines = code.split('\n');
    
    for (let i = 0; i < lines.length; i++) {
      const line = lines[i].trim();
      
      // تتبع المتغيرات
      if (line.startsWith('متغير')) {
        const match = line.match(/متغير\s+(\w+)\s*=\s*(.+)؛/);
        if (match) {
          this.variables.set(match[1], this.evaluate(match[2]));
          steps.push({
            line: i + 1,
            type: 'variable',
            name: match[1],
            value: this.variables.get(match[1])
          });
        }
      }
      
      // تتبع الطباعة
      if (line.startsWith('اطبع')) {
        steps.push({
          line: i + 1,
          type: 'output',
          content: this.evaluatePrint(line)
        });
      }
    }
    
    return steps;
  }
}
```
